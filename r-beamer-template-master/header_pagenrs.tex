% Setup the footline of each slide.
% This requires R code output to have out.height="85%", so that the graphs won't be truncated by the footnote of each slide.
% Code source: https://tex.stackexchange.com/questions/51665/appearance-of-the-name-of-the-author-in-beamer
\setbeamertemplate{footline}{%
  \leavevmode%
  \hbox{%
  \begin{beamercolorbox}[wd=.333333\paperwidth,ht=3.25ex,dp=1ex,center]{author in head/foot}%
    \usebeamerfont{author in head/foot}\insertauthor
  \end{beamercolorbox}%
  \begin{beamercolorbox}[wd=.333333\paperwidth,ht=3.25ex,dp=1ex,center]{title in head/foot}%
    \usebeamerfont{title in head/foot}\insertshortinstitute
  \end{beamercolorbox}%
  \begin{beamercolorbox}[wd=.333333\paperwidth,ht=3.25ex,dp=1ex,right]{date in head/foot}%
    \usebeamerfont{date in head/foot}\today\hspace*{2em}
    \insertframenumber{} \hspace*{2ex} 
  \end{beamercolorbox}}%
  \vskip0pt%
}
